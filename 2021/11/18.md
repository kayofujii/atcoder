A,B,C=map(int,input().split())
 
if C == 0:
  print('=')
  exit()
 
if C % 2 == 0:
  if abs(A) < abs(B):
    print('<')
  elif abs(A) > abs(B):
    print('>')
  else:
    print('=')
else:
  if 0<=A:
    if 0<=B:
      if abs(A) < abs(B):
        print('<')
      elif abs(A) > abs(B):
        print('>')
      else:
        print('=')
    elif B<0:
      print('>')
  elif A<0:
    if 0<=B:
      print('<')
    elif B<0:
      if abs(A) < abs(B):
        print('<')
      elif abs(A) > abs(B):
        print('>')
      else:
        print('=')
