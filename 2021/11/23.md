S=list(input())
S.reverse()
ans_list = []
 
for i in range(len(S)):
  if S[i] == '6':
    ans_list.append('9')
  elif S[i] == '9':
    ans_list.append('6')
  else:
    ans_list.append(S[i])
 
print(''.join(ans_list))

from collections import Counter
A=list(map(int,input().split()))

if len(set(A)) == 1:
  print(A[0])
elif len(set(A)) == 2:
  dup = [key for key,val in Counter(A).items() if val > 1]
  ans = list(set(A)-set(dup))
  print(ans[0])
else:
  print('0')
